/*
this will return the num'th fibonacci number
also this is just a test for comments in codeblocks
*/
function fibonacci(num){
  var a = 1, b = 0, temp;

  while (num >= 0){
    temp = a;
    a = a + b;
    b = temp;
    num--;
  }

  return b;
}

let i = 0;

while(true){
	if (i == 69){
		console.log("69 lol");
	}

	fibonacci(i);
	i++;
}
